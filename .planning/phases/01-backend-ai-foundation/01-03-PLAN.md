# Plan 01-03: AI Detection Workers — YOLOv8 + ByteTrack Pipeline

## Objective
Implement the complete multiprocessing AI pipeline: one Python process per camera, each running YOLOv8n + ByteTrack with every-3rd-frame processing. Each worker maintains its own density state machine (D-08: 5 sustained frames + 60s cooldown), computes person counts, and POSTs results to the FastAPI backend. A supervisor process manages worker lifecycle with heartbeat-based fault detection and automatic restart.

    ```python
    """
    CrowdDetector: Wraps YOLOv8n + ByteTrack for single-camera inference.
    Each instance is created INSIDE a child process (never shared across processes).
    """
    from ultralytics import YOLO


    class CrowdDetector:
        """
        Initializes YOLOv8n model with ByteTrack tracker.
        MUST be instantiated inside the worker process, not before fork/spawn.
        """

        MODEL_PATH = "yolov8n.pt"
        TRACKER_CONFIG = "bytetrack.yaml"
        INFERENCE_SIZE = 416  # imgsz=416 for speed optimization

        def __init__(self, max_age: int = 10):
            self.model = YOLO(self.MODEL_PATH)
            self.max_age = max_age
            self.frame_count = 0

        def process_frame(self, frame) -> int | None:
            """
            Process a single video frame.
            Returns person_count (int) if frame was processed,
            or None if frame was skipped (D-03: every 3rd frame).
            """
            self.frame_count += 1

            # D-03: skip every frame where count % 3 != 0
            # i.e., process frames 3, 6, 9, 12... (every 3rd frame)
            if self.frame_count % 3 != 0:
                return None

            results = self.model.track(
                frame,
                tracker=self.TRACKER_CONFIG,
                persist=True,
                imgsz=self.INFERENCE_SIZE,
                verbose=False,
            )

            if results and results[0].boxes is not None:
                return len(results[0].boxes)
            return 0
    ```

    ```python
    """
    camera_worker: Runs in a dedicated child process per camera.
    Reads frames -> detects crowd -> maintains alert state -> POSTs to API.
    """
    import time
    import multiprocessing
    import httpx
    import cv2
    from workers.detector import CrowdDetector

    API_BASE_URL = "http://localhost:8000"
    API_KEY = ""  # Loaded from env in supervisor.py before spawning


    def compute_density_level(person_count: int, low_thresh: int, high_thresh: int) -> str:
        """Maps person count to density label based on per-camera thresholds. (D-07)"""
        if person_count < low_thresh:
            return "low"
        elif person_count < high_thresh:
            return "medium"
        else:
            return "high"


    def camera_worker(
        cam_config: dict,
        heartbeat: multiprocessing.Value,
        api_key: str,
    ) -> None:
        """
        Main loop for a single camera worker process.
        cam_config keys: id, source, low_thresh, high_thresh, max_age
        heartbeat: shared multiprocessing.Value('d') updated each successful frame
        """
        camera_id = cam_config["id"]
        low_thresh = cam_config.get("low_thresh", 30)
        high_thresh = cam_config.get("high_thresh", 75)
        max_age = cam_config.get("max_age", 10)

        detector = CrowdDetector(max_age=max_age)
        cap = cv2.VideoCapture(cam_config["source"])

        # D-08: Alert state machine
        consecutive_high_frames = 0
        HIGH_FRAME_THRESHOLD = 5       # Must sustain for 5 frames
        ALERT_COOLDOWN_SECS = 60       # 60s between alerts for same camera
        last_alert_time = 0.0

        try:
            while True:
                ret, frame = cap.read()
                if not ret:
                    # Stream dropped — wait and reconnect
                    time.sleep(2)
                    cap.release()
                    cap = cv2.VideoCapture(cam_config["source"])
                    continue

                heartbeat.value = time.time()  # D: keep heartbeat alive

                person_count = detector.process_frame(frame)
                if person_count is None:
                    continue  # Frame skipped (D-03)

                density_level = compute_density_level(person_count, low_thresh, high_thresh)

                # D-08: Track consecutive high-density frames
                if density_level == "high":
                    consecutive_high_frames += 1
                else:
                    consecutive_high_frames = 0

                now = time.time()
                trigger_alert = (
                    consecutive_high_frames >= HIGH_FRAME_THRESHOLD
                    and (now - last_alert_time) >= ALERT_COOLDOWN_SECS
                )
                if trigger_alert:
                    last_alert_time = now
                    consecutive_high_frames = 0  # Reset after alert fires

                # D-09: POST crowd snapshot to FastAPI
                payload = {
                    "camera_id": camera_id,
                    "person_count": person_count,
                    "density_level": density_level,
                    "trigger_alert": trigger_alert,
                    "alert_threshold": high_thresh,
                }
                try:
                    httpx.post(
                        f"{API_BASE_URL}/crowd/update",
                        json=payload,
                        headers={"X-API-Key": api_key},
                        timeout=2.0,
                    )
                except httpx.RequestError:
                    pass  # Non-blocking: skip POST failures silently

        finally:
            cap.release()  # Always release camera on exit
    ```

    ```python
    """
    Supervisor: Reads cameras.json, spawns one process per active camera,
    and monitors processes with heartbeat-based fault detection + auto-restart.

    Usage: python workers/supervisor.py
    """
    import json
    import multiprocessing
    import os
    import time
    import signal
    import sys
    from workers.processor import camera_worker

    CAMERAS_CONFIG_PATH = "cameras.json"
    HEARTBEAT_TIMEOUT_SECS = 30   # If no heartbeat update in 30s, restart
    MONITOR_INTERVAL_SECS = 5     # Check worker health every 5 seconds
    RESTART_DELAY_SECS = 3        # Delay before restarting a dead worker


    def load_cameras() -> list[dict]:
        with open(CAMERAS_CONFIG_PATH, "r") as f:
            data = json.load(f)
        return [cam for cam in data["cameras"] if cam.get("active", True)]


    def spawn_worker(cam_config: dict, api_key: str) -> tuple:
        """Spawn a new camera worker process. Returns (process, heartbeat_value)."""
        heartbeat = multiprocessing.Value("d", time.time())
        p = multiprocessing.Process(
            target=camera_worker,
            args=(cam_config, heartbeat, api_key),
            name=f"worker-{cam_config['id']}",
            daemon=False,
        )
        p.start()
        print(f"[Supervisor] Started worker for camera {cam_config['id']} (PID {p.pid})")
        return p, heartbeat


    def run_supervisor():
        api_key = os.environ.get("API_KEY", "")
        cameras = load_cameras()
        print(f"[Supervisor] Loaded {len(cameras)} active cameras from {CAMERAS_CONFIG_PATH}")

        workers: dict[str, dict] = {}
        for cam in cameras:
            p, hb = spawn_worker(cam, api_key)
            workers[cam["id"]] = {"process": p, "heartbeat": hb, "config": cam}

        def handle_shutdown(signum, frame):
            print("[Supervisor] Shutdown signal received — terminating workers")
            for cam_id, w in workers.items():
                w["process"].terminate()
                w["process"].join(timeout=5)
            sys.exit(0)

        signal.signal(signal.SIGINT, handle_shutdown)
        signal.signal(signal.SIGTERM, handle_shutdown)

        while True:
            time.sleep(MONITOR_INTERVAL_SECS)
            for cam_id, w in workers.items():
                p = w["process"]
                hb_age = time.time() - w["heartbeat"].value

                if not p.is_alive() or hb_age > HEARTBEAT_TIMEOUT_SECS:
                    reason = "crashed" if not p.is_alive() else f"heartbeat stale ({hb_age:.0f}s)"
                    print(f"[Supervisor] Worker {cam_id} {reason} — restarting in {RESTART_DELAY_SECS}s")
                    p.terminate()
                    p.join()
                    time.sleep(RESTART_DELAY_SECS)
                    new_p, new_hb = spawn_worker(w["config"], api_key)
                    workers[cam_id]["process"] = new_p
                    workers[cam_id]["heartbeat"] = new_hb


    if __name__ == "__main__":
        # CRITICAL: must use 'spawn' for PyTorch/CUDA in child processes
        multiprocessing.set_start_method("spawn", force=True)
        run_supervisor()
    ```
